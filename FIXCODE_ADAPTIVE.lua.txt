local function GetScreenSize()
    local mouse = localPlayer:GetMouse()
    local width = mouse.ViewSizeX or 1920
    local height = mouse.ViewSizeY or 1080
    return {width = width, height = height}
end

local function GetDeviceType()
    local screenSize = GetScreenSize()
    local width = screenSize.width
    
    if width <= 480 then
        return "mobile"
    elseif width <= 768 then
        return "tablet"
    elseif width <= 1280 then
        return "laptop"
    else
        return "desktop"
    end
end

local function GetScreenOrientation()
    local screenSize = GetScreenSize()
    return screenSize.width > screenSize.height and "landscape" or "portrait"
end

local function ScanCompleteWorkspace()
    local data = {
        remoteEvents = {},
        remoteFunctions = {},
        models = {},
        folders = {},
        parts = {},
        scripts = {},
        localScripts = {},
        moduleScripts = {}
    }
    
    local function DeepScan(folder, path, maxDepth, currentDepth)
        if currentDepth > maxDepth then return end
        
        for _, obj in pairs(folder:GetChildren()) do
            pcall(function()
                local currentPath = path .. "/" .. obj.Name
                
                if obj:IsA("RemoteEvent") then
                    table.insert(data.remoteEvents, {name = obj.Name, path = currentPath, parent = folder.Name, type = "RemoteEvent"})
                elseif obj:IsA("RemoteFunction") then
                    table.insert(data.remoteFunctions, {name = obj.Name, path = currentPath, parent = folder.Name, type = "RemoteFunction"})
                elseif obj:IsA("Model") then
                    table.insert(data.models, {name = obj.Name, path = currentPath, childCount = #obj:GetChildren()})
                elseif obj:IsA("Folder") then
                    table.insert(data.folders, {name = obj.Name, path = currentPath, childCount = #obj:GetChildren()})
                elseif obj:IsA("Part") or obj:IsA("BasePart") then
                    table.insert(data.parts, {name = obj.Name, path = currentPath})
                elseif obj:IsA("LocalScript") then
                    table.insert(data.localScripts, {name = obj.Name, path = currentPath})
                elseif obj:IsA("Script") then
                    table.insert(data.scripts, {name = obj.Name, path = currentPath})
                elseif obj:IsA("ModuleScript") then
                    table.insert(data.moduleScripts, {name = obj.Name, path = currentPath})
                end
                
                if obj:IsA("Folder") or obj:IsA("Model") then
                    DeepScan(obj, currentPath, maxDepth, currentDepth + 1)
                end
            end)
        end
    end
    
    DeepScan(workspace, "Workspace", 8, 0)
    DeepScan(ReplicatedStorage, "ReplicatedStorage", 7, 0)
    
    data.totalRemotes = #data.remoteEvents + #data.remoteFunctions
    data.totalModels = #data.models
    data.totalScripts = #data.scripts + #data.localScripts + #data.moduleScripts
    data.totalParts = #data.parts
    
    return data
end

local function CreateOrientationHandler()
    local handler = {}
    local currentOrientation = GetScreenOrientation()
    
    function handler:OnOrientationChange(callback)
        RunService.RenderStepped:Connect(function()
            local newOrientation = GetScreenOrientation()
            if newOrientation ~= currentOrientation then
                currentOrientation = newOrientation
                callback(currentOrientation)
                HandleScreenResize()
            end
        end)
    end
    
    return handler
end

-- Performance monitoring for different screen sizes
local function CreatePerformanceOptimizer()
    local optimizer = {}
    local metrics = {fps = 0, memory = 0, threadCount = 0}
    
    function optimizer:MonitorPerformance()
        local frameCount = 0
        local lastTime = tick()
        
        RunService.Heartbeat:Connect(function()
            frameCount = frameCount + 1
            local currentTime = tick()
            
            if currentTime - lastTime >= 1 then
                metrics.fps = frameCount
                frameCount = 0
                lastTime = currentTime
            end
        end)
    end
    
    function optimizer:OptimizeForLowPerformance()
        if metrics.fps < 30 then
            print("[OPTIMIZATION] FPS low, disabling visual effects")
            for _, button in pairs(ScrollFrame:GetChildren()) do
                if button:IsA("TextButton") then
                    button.BorderSizePixel = 0
                end
            end
        end
    end
    
    function optimizer:GetMetrics()
        return metrics
    end
    
    return optimizer
end

-- Multi-resolution support
local function CreateResolutionManager()
    local manager = {}
    
    local resolutions = {
        {name = "320p", width = 320, height = 568, ratio = "9:16"},
        {name = "480p", width = 480, height = 854, ratio = "9:16"},
        {name = "720p", width = 720, height = 1280, ratio = "9:16"},
        {name = "768p", width = 768, height = 1024, ratio = "4:3"},
        {name = "1080p", width = 1080, height = 1920, ratio = "9:16"},
        {name = "1280p", width = 1280, height = 720, ratio = "16:9"},
        {name = "1920p", width = 1920, height = 1080, ratio = "16:9"},
        {name = "2560p", width = 2560, height = 1440, ratio = "16:9"}
    }
    
    function manager:GetClosestResolution()
        local screenSize = GetScreenSize()
        local minDiff = math.huge
        local closest = resolutions[1]
        
        for _, res in ipairs(resolutions) do
            local diff = math.abs(res.width - screenSize.width) + math.abs(res.height - screenSize.height)
            if diff < minDiff then
                minDiff = diff
                closest = res
            end

-- State management system
local function CreateStateManager()
    local state = {}
    local stateData = {}
    local subscribers = {}
    
    function state:SetState(key, value)
        local oldValue = stateData[key]
        stateData[key] = value
        
        if subscribers[key] then
            for _, callback in ipairs(subscribers[key]) do
                callback(value, oldValue)
            end
        end
    end
    
    function state:GetState(key)
        return stateData[key]
    end
    
    function state:Subscribe(key, callback)
        if not subscribers[key] then
            subscribers[key] = {}
        end
        table.insert(subscribers[key], callback)
    end
    
    function state:UnsubscribeAll()
        subscribers = {}
    end
    
    return state
end

local StateManager = CreateStateManager()

        local currentTime = tick()
        
        for i, anim in ipairs(animations) do
            if anim.active then

                local progress = math.min(1, elapsed / anim.duration)
                
                for prop, endValue in pairs(anim.properties) do
                    -- Apply smooth interpolation
                end
                
                if progress >= 1 then
                    anim.active = false
                    table.remove(animations, i)
                end
            end
        end
    end
    
    return animator

-- Utility functions for common tasks
local function CreateUtilityFunctions()
    local utils = {}
    
    function utils:FormatBytes(bytes)
        if bytes < 1024 then
            return bytes .. "B"
        elseif bytes < 1024 * 1024 then
            return math.floor(bytes / 1024) .. "KB"
        else
            return math.floor(bytes / (1024 * 1024)) .. "MB"
        end
    end
    
    function utils:FormatTime(seconds)
        local mins = math.floor(seconds / 60)
        local secs = seconds % 60
        return mins .. "m " .. secs .. "s"
    end
    
    function utils:Debounce(func, delay)
        local timeout = nil
        return function(...)
            if timeout then
                RunService:CancelScheduledEvent(timeout)
            end
            timeout = RunService:ScheduleEvent(delay, func)
        end
    end
    
    function utils:Throttle(func, delay)
        local last = 0
        return function(...)
            local now = tick()
            if now - last >= delay then
                last = now
                return func(...)
            end
        end
    end
    
    return utils
end

local Utils = CreateUtilityFunctions()

-- Event system
local function CreateEventBus()
    local events = {}
    local listeners = {}
    
    function events:Emit(eventName, ...)
        if listeners[eventName] then
            for _, callback in ipairs(listeners[eventName]) do
                callback(...)
            end
        end
    end
    
    function events:On(eventName, callback)
        if not listeners[eventName] then
            listeners[eventName] = {}
        end
        table.insert(listeners[eventName], callback)
    end
    
    function events:Off(eventName, callback)
        if listeners[eventName] then
            for i, cb in ipairs(listeners[eventName]) do
                if cb == callback then
                    table.remove(listeners[eventName], i)
                    break
                end
            end
        end
    end
    
    return events
end

    local logger = {}

        local log = "[INFO] " .. message

    end
    
    function logger:Warning(message)
        local log = "[WARNING] " .. message
        table.insert(logs, log)
        print(log)
    end
    
    function logger:Error(message)
        local log = "[ERROR] " .. message
        table.insert(logs, log)
        print(log)
    end
    
    function logger:Debug(message)
        if ConfigManager:GetConfig().debugMode then
            local log = "[DEBUG] " .. message
            table.insert(logs, log)
            print(log)
        end
    end
    
    function logger:GetLogs()
        return logs
    end
    
    function logger:ClearLogs()
        logs = {}
    end
    
    return logger
end
