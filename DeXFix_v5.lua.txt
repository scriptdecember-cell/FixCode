local function Tw(o,p,t,s)
    Tween:Create(o,TweenInfo.new(t or .18,s or Enum.EasingStyle.Quart,Enum.EasingDirection.Out),p):Play()
end
local function Rnd(p,r)
    local c=Instance.new("UICorner")
    c.CornerRadius=UDim.new(0,r or 4)
    c.Parent=p
    return c
end
local function Stk(p,c,t)
    local s=Instance.new("UIStroke")
    s.Color=c
    s.Thickness=t or 1
    s.Parent=p
    return s
end
local function Frm(par,col,sz,pos,zi,clip)
    local f=Instance.new("Frame",par)
    f.BackgroundColor3=col or K.BG
    f.BorderSizePixel=0
    if sz   then f.Size=sz     end
    if pos  then f.Position=pos end
    if zi   then f.ZIndex=zi   end
    if clip then f.ClipsDescendants=true end
    return f
end
local function Lbl(par,txt,fsz,col,font,xa)
    local l=Instance.new("TextLabel",par)
    l.Text=txt or ""
    l.TextSize=fsz or S(11)
    l.TextColor3=col or K.Txt

    l.Size=UDim2.new(1,0,1,0)
    l.TextXAlignment=xa or Enum.TextXAlignment.Left
    l.TextYAlignment=Enum.TextYAlignment.Center

end

local function SwitchTab(idx)
    if idx==activeTab then return end
    local old=activeTab
    activeTab=idx
    Tw(tabBtns[old],{BackgroundColor3=K.BG2})
    tabBtns[old].TextColor3=K.Sub
    for _,s in pairs(tabBtns[old]:GetChildren()) do
        if s:IsA("UIStroke") then s:Destroy() end
    end
    Tw(tabBtns[idx],{BackgroundColor3=K.AccDk})
    tabBtns[idx].TextColor3=K.Acc
    Stk(tabBtns[idx],K.AccD,1)
    tabPages[old].Visible=false
    tabPages[idx].Visible=true
end

for i=1,#tabNames do
    local ii=i
    tabBtns[i].MouseButton1Click:Connect(function() SwitchTab(ii) end)
    tabBtns[i].TouchTap:Connect(function() SwitchTab(ii) end)
end

local function AddHistory(jobId, playing, maxPlayers, ping)
    for _,h in ipairs(serverHistory) do
        if h.id==jobId then return end
    end
    table.insert(serverHistory,1,{
        id=jobId,
        playing=playing,
        max=maxPlayers,
        ping=ping,
        time=os.time(),
    })
    if #serverHistory>20 then
        table.remove(serverHistory,#serverHistory)
    end
end

local function IsFav(jobId)
    for _,f in ipairs(serverFavs) do
        if f.id==jobId then return true end
    end
    return false
end

local function ToggleFav(jobId,playing,maxPlayers,ping)
    for i,f in ipairs(serverFavs) do
        if f.id==jobId then
            table.remove(serverFavs,i)
            return false
        end
    end
    table.insert(serverFavs,{id=jobId,playing=playing,max=maxPlayers,ping=ping})
    return true
end

local function ScanServers()
    if scanning then return end
    scanning=true
    ClearScroll()
    SetStat("SCAN",K.Yel)
    FindL.Text="SCANNING..."
    CountLbl.Text="[ · · · ]"
    InfoLeft.Text="> scanning..."

    local ph=Frm(Scrl,K.Panel,UDim2.new(1,-S(2),0,math.max(SERV_SCR_H-S(10),S(80))))
    ph.LayoutOrder=0
    Rnd(ph,S(4))
    Stk(ph,K.Line,1)
    local dots={"[·  ]","[·· ]","[···]"}
    local di=1
    local anm=Run.Heartbeat:Connect(function()
        di=di%3+1
        if ph and ph.Parent then
            for _,c in pairs(ph:GetChildren()) do
                if c:IsA("TextLabel") and c.TextSize>S(14) then
                    c.Text=dots[di]
                end
            end
        end
    end)
    local ic=Lbl(ph,dots[1],S(18),K.AccD,Enum.Font.Code,Enum.TextXAlignment.Center)
    ic.Size=UDim2.new(1,0,0.4,0)
    ic.Position=UDim2.new(0,0,0.12,0)
    ic.TextYAlignment=Enum.TextYAlignment.Center
    local sl=Lbl(ph,"scanning servers...",S(10),K.Sub,Enum.Font.Code,Enum.TextXAlignment.Center)
    sl.Size=UDim2.new(1,0,0,S(16))
    sl.Position=UDim2.new(0,0,0.62,0)

    task.spawn(function()
        local found,cursor,pages={},nil,0
        local totalSeen=0
        repeat
            pages=pages+1
            local url="https://games.roblox.com/v1/games/"..game.PlaceId..
                "/servers/Public?sortOrder=Asc&limit=100"
            if cursor then url=url.."&cursor="..cursor end
            local ok,raw=pcall(function() return game:HttpGet(url) end)
            if not ok then break end
            local ok2,data=pcall(function() return HttpService:JSONDecode(raw) end)
            if not ok2 or not data or not data.data then break end
            for _,s in ipairs(data.data) do
                totalSeen=totalSeen+1
                local pct2=s.playing/math.max(s.maxPlayers,1)
                if pct2<0.15 then scanStats.empty=scanStats.empty+1
                elseif pct2<0.5 then scanStats.half=scanStats.half+1
                else scanStats.full=scanStats.full+1end
                if s.playing<=maxP and s.id~=game.JobId then
                    table.insert(found,{
                        p=s.playing,
                        m=s.maxPlayers,
                        id=s.id,
                        ping=math.random(14,150),
                    })
                end
                if #found>=maxResults then break end
            end
            cursor=data.nextPageCursor
        until #found>=maxResults or not cursor or pages>=5

        scanStats.total=scanStats.total+totalSeen
        scanStats.scans=scanStats.scans+1

        if sortMode=="ping" then
            table.sort(found,function(a,b) return a.ping<b.ping end)
        else
            table.sort(found,function(a,b) return a.p<b.p end)
        end

        anm:Disconnect()
        if ph and ph.Parent then ph:Destroy() end

        if #found==0 then
            ShowMsg("NO RESULTS","try increasing the player filter")
            SetStat("EMPTY",K.Red)
            Toast(">> no results — try max="..maxP.." higher",K.Red,4)
        else
            for i,s in ipairs(found) do
                AddServerCard(i,s.p,s.m,s.id,s.ping)
                task.wait(0.016)
            end
            SetStat("OK",K.Acc)
            CountLbl.Text="[ "..#found.." found ]"
            InfoLeft.Text="> server_list.txt"
            Toast(">> found "..#found.." servers  ≤"..maxP.." players",K.Acc,3)
        end
        FindL.Text="SCAN SERVERS"
        scanning=false
    end)
end

local function RandomHop()
    SetStat("HOP",K.Red)
    HopL.Text="HOPPING..."
    Toast(">> random server hop...",K.Red,2)
    task.spawn(function()
        local ok,raw=pcall(function()
            return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId..
                "/servers/Public?sortOrder=Desc&limit=100")
        end)
        local picked=false
        if ok then
            local ok2,data=pcall(function() return HttpService:JSONDecode(raw) end)
            if ok2 and data and data.data then
                local valid={}
                for _,s in ipairs(data.data) do
                    if s.id~=game.JobId and s.playing<s.maxPlayers then
                        table.insert(valid,s)
                    end
                end
                if #valid>0 then
                    local pick=valid[math.random(1,#valid)]
                    AddHistory(pick.id,pick.playing,pick.maxPlayers,math.random(14,150))
                    scanStats.hops=scanStats.hops+1
                    task.wait(0.4)
                    pcall(function()
                        TeleportService:TeleportToPlaceInstance(game.PlaceId,pick.id,LP)
                    end)
                    picked=true
                end
            end
        end
        if not picked then
            task.wait(0.4)
            pcall(function() TeleportService:Teleport(game.PlaceId,LP) end)
        end
        HopL.Text="RANDOM HOP"
        SetStat("IDLE",K.Sub)
    end)
end

                Toast(">> auto-scan every "..autoScanDelay.."s",K.Yel,3)
            else
                if autoScanConn then autoScanConn:Disconnect() autoScanConn=nil end
                Toast(">> auto-scan disabled",K.Sub,2)
            end
        end,K.Yel)

    SliderSetting("Scan Interval","Seconds between auto-scans",10,120,
        function() return autoScanDelay end,
        function(v) autoScanDelay=v end,"s",K.Yel)

    SecHdr("Info",K.Sub)

    local InfoCard=Frm(sc,K.Panel,UDim2.new(1,-S(2),0,S(76)))
    Rnd(InfoCard,S(4)) Stk(InfoCard,K.Line,1)
    local infoLines={
        "DeXFix v5.0  //  Server Explorer",
        "Private server uses ReserveServer+TeleportAsync",
        "♥ = save server to favourites",
        "History tracks last 20 joined servers",
    }
    for ii,ln in ipairs(infoLines) do
        local il=Lbl(InfoCard,ln,S(9),K.Sub,Enum.Font.Code)
        il.Size=UDim2.new(1,-S(12),0,S(15))
        il.Position=UDim2.new(0,S(8),0,(ii-1)*S(17)+S(4))
    end
end

local dS,dWP
HEAD.InputBegan:Connect(function(i)
    if i.UserInputType==Enum.UserInputType.MouseButton1
    or i.UserInputType==Enum.UserInputType.Touch then
        dS=i.Position dWP=Win.Position
    end
end)
UIS.InputChanged:Connect(function(i)
    if dS and (i.UserInputType==Enum.UserInputType.MouseMovement
    or i.UserInputType==Enum.UserInputType.Touch) then
        local d=i.Position-dS
        local nx=math.clamp(dWP.X.Offset+d.X,0,SW-W)
        local ny=math.clamp(dWP.Y.Offset+d.Y,0,SH-HEAD_H)
        Win.Position=UDim2.new(0,nx,0,ny)
    end
end)
UIS.InputEnded:Connect(function(i)
    if i.UserInputType==Enum.UserInputType.MouseButton1
    or i.UserInputType==Enum.UserInputType.Touch then dS=nil end
end)

UIS.InputBegan:Connect(function(input,gpe)
    if gpe then return end
    if input.KeyCode==Enum.KeyCode.F5 then ScanServers() end
    if input.KeyCode==Enum.KeyCode.F6 then SoloSession() end
    if input.KeyCode==Enum.KeyCode.F7 then RandomHop()   end
    if input.KeyCode==Enum.KeyCode.F8 then
        minimized=not minimized
        Tw(Win,{Size=minimized and UDim2.new(0,W,0,HEAD_H+2) or UDim2.new(0,W,0,H)},0.28,Enum.EasingStyle.Back)
        for _,p in pairs(tabPages) do p.Visible=not minimized and p==tabPages[activeTab] end
        TabBar.Visible=not minimized
        MinBtn.Text=minimized and "[+]" or "[−]"
    end
end)
