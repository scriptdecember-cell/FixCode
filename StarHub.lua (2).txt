local function doSpin(bag)
    local s = pcall(function() SpinRemote:InvokeServer(bag) end)
    if s then State.SpinCount = State.SpinCount + 1 end
    return s
end

local function doReset()
    if not ResetRemote then return false end
    local s = pcall(function() ResetRemote:InvokeServer() end)
    if s then State.ResetCount = State.ResetCount + 1 end
    return s
end

local function redeemAllCodes(notifCb)
    local ok, fail = 0, 0
    for _, code in ipairs(Codes) do
        local s = pcall(function() CodeRemote:InvokeServer(code) end)
        if s then ok = ok + 1 else fail = fail + 1 end
        if notifCb then notifCb(code, ok, fail) end
        task.wait(0.3)
    end
    return ok, fail
end

local function updateStats()
    StatsLabel:Set("Spins: " .. State.SpinCount .. "  |  Resets: " .. State.ResetCount .. "  |  Cycles: " .. State.CycleCount)
end

SpinTab:CreateToggle({
    Name = "Auto Cycle",
    CurrentValue = false,
    Flag = "AutoCycle",
    Callback = function(val)
        cycleRunning = val
        if val then
            CycleStatusLabel:Set("Status: Running...")
            task.spawn(function()
                while cycleRunning do
                    CycleStatusLabel:Set("Redeeming codes...")
                    redeemAllCodes(nil)
                    task.wait(1)

                    CycleStatusLabel:Set("Spinning Bag 1...")
                    for _ = 1, 200 do
                        if not cycleRunning then break end
                        doSpin(1)
                        updateStats()
                        task.wait(0.08)
                    end

                    CycleStatusLabel:Set("Spinning Bag 2...")
                    for _ = 1, 200 do
                        if not cycleRunning then break end
                        doSpin(2)
                        updateStats()
                        task.wait(0.08)
                    end

                    CycleStatusLabel:Set("Resetting...")
                    doReset()
                    State.CycleCount = State.CycleCount + 1
                    updateStats()
                    task.wait(2)
                end
                CycleStatusLabel:Set("Status: Stopped")
            end)
        else
            CycleStatusLabel:Set("Status: Stopped")
        end
    end,
})
