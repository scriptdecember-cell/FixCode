local function ScanRemotes()
    for _, v in ipairs(game:GetDescendants()) do
        if v:IsA("RemoteEvent") or v:IsA("RemoteFunction") then
            local n = v.Name:lower()
            if n:find("parry") or n:find("block") or n:find("guard") then
                table.insert(ParryRemotes, v)
            end
            if n:find("gen") then
                table.insert(GenRemotes, v)
            end
            if n:find("attack") or n:find("m1") or n:find("hit") then
                table.insert(AttackRemotes, v)
            end
        end
    end
end
task.delay(3, ScanRemotes)

local function FindObjects(keys)
    local found = {}
    for _, key in ipairs(keys) do
        for _, v in ipairs(workspace:GetDescendants()) do
            if v.Name:find(key) and v:IsA("BasePart") then
                found[#found+1] = v
            end
        end
    end
    return found
end

local function NearestObject(keys)
    if not Root then return nil end
    local best, bestDist = nil, math.huge
    for _, v in ipairs(FindObjects(keys)) do
        local d = (v.Position - Root.Position).Magnitude
        if d < bestDist then bestDist = d; best = v end
    end
    return best
end

local function SafeTP(pos)
    if Root then
        Root.CFrame = CFrame.new(pos + Vector3.new(0, 4, 0))
    end
end

local function FireAll(list, ...)
    for _, r in ipairs(list) do
        pcall(function() r:FireServer(...) end)
    end
end

local function InteractObject(obj)
    if not obj then return end
    local cd = obj:FindFirstChildWhichIsA("ClickDetector")
    if cd then pcall(fireclickdetector, cd) end
    local pp = obj:FindFirstChildWhichIsA("ProximityPrompt")
    if pp then pcall(fireproximityprompt, pp) end
    FireAll(GenRemotes, obj)
end

local function MakeHighlight(adornee, fill, outline)
    local h = Instance.new("Highlight")
    h.DepthMode           = Enum.HighlightDepthMode.AlwaysOnTop
    h.FillColor           = fill    or Color3.fromRGB(255,255,255)
    h.OutlineColor        = outline or Color3.fromRGB(255,255,255)
    h.FillTransparency    = 0.45
    h.OutlineTransparency = 0
    h.Adornee             = adornee
    h.Parent              = game.CoreGui
    return h
end

local function MakeBillboard(adornee, text, color, width)
    local bb = Instance.new("BillboardGui")
    bb.AlwaysOnTop  = true
    bb.Size         = UDim2.new(0, width or 150, 0, 26)
    bb.StudsOffset  = Vector3.new(0, 3.5, 0)
    bb.Adornee      = adornee
    bb.Parent       = game.CoreGui

    local bg = Instance.new("Frame", bb)
    bg.Size                   = UDim2.new(1,0,1,0)
    bg.BackgroundColor3       = Color3.fromRGB(12,12,12)
    bg.BackgroundTransparency = 0.25
    bg.BorderSizePixel        = 0
    Instance.new("UICorner", bg).CornerRadius = UDim.new(0,6)

    local lbl = Instance.new("TextLabel", bg)
    lbl.Size                   = UDim2.new(1,-6,1,0)
    lbl.Position               = UDim2.new(0,3,0,0)
    lbl.BackgroundTransparency = 1
    lbl.Text                   = text
    lbl.TextColor3             = color or Color3.fromRGB(255,255,255)
    lbl.TextStrokeTransparency = 0.4
    lbl.TextScaled             = true
    lbl.Font                   = Enum.Font.GothamBold
    return bb, lbl
end

local function ClearESP(key)
    if ESPHighlights[key] then
        for _, h in ipairs(ESPHighlights[key]) do pcall(function() h:Destroy() end) end
        ESPHighlights[key] = nil
    end
    if ESPBillboards[key] then
        for _, b in ipairs(ESPBillboards[key]) do pcall(function() b:Destroy() end) end
        ESPBillboards[key] = nil
    end
end

local function BuildPlayerESP(key, fill, outline, label, labelColor)
    ClearESP(key)
    ESPHighlights[key] = {}
    ESPBillboards[key] = {}

    local function Apply(p)
        if p == LP then return end
        local char = p.Character
        if not char then return end
        local hl = MakeHighlight(char, fill, outline)
        table.insert(ESPHighlights[key], hl)
        local root = char:FindFirstChild("HumanoidRootPart")
        if root then
            local bb, lbl = MakeBillboard(root, label .. p.Name, labelColor)
            table.insert(ESPBillboards[key], bb)
            RunService.Heartbeat:Connect(function()
                if not S[key] or not Root or not root or not root.Parent then return end
                local dist = math.floor((root.Position - Root.Position).Magnitude)
                lbl.Text = label .. p.Name .. " [" .. dist .. "m]"
            end)
        end
        p.CharacterRemoving:Connect(function() pcall(function() hl:Destroy() end) end)
        p.CharacterAdded:Connect(function() task.wait(0.5); if S[key] then Apply(p) end end)
    end

    for _, p in ipairs(Players:GetPlayers()) do Apply(p) end
    Players.PlayerAdded:Connect(function(p)
        p.CharacterAdded:Connect(function() task.wait(0.5); if S[key] then Apply(p) end end)
    end)
end

local function BuildObjectESP(key, names, fill, outline, label, labelColor)
    ClearESP(key)
    ESPHighlights[key] = {}
    ESPBillboards[key] = {}
    for _, obj in ipairs(FindObjects(names)) do
        local hl = MakeHighlight(obj, fill, outline)
        table.insert(ESPHighlights[key], hl)
        local bb, lbl = MakeBillboard(obj, label, labelColor)
        table.insert(ESPBillboards[key], bb)
        RunService.Heartbeat:Connect(function()
            if not S[key] or not Root or not obj or not obj.Parent then return end
            lbl.Text = label .. " [" .. math.floor((obj.Position - Root.Position).Magnitude) .. "m]"
        end)
    end
end

local function BuildKillerHUD()
    if KillerHUDGui then KillerHUDGui:Destroy() end
    local sg = Instance.new("ScreenGui", game.CoreGui)
    sg.Name         = "FixHubKillerHUD"
    sg.ResetOnSpawn = false
    KillerHUDGui    = sg

    local outer = Instance.new("Frame", sg)
    outer.AnchorPoint           = Vector2.new(0.5, 0)
    outer.Position              = UDim2.new(0.5, 0, 0.04, 0)
    outer.Size                  = UDim2.new(0, 300, 0, 46)
    outer.BackgroundColor3      = Color3.fromRGB(10,10,10)
    outer.BackgroundTransparency = 0.2
    outer.BorderSizePixel       = 0
    outer.Visible               = false
    Instance.new("UICorner", outer).CornerRadius = UDim.new(0,10)

    local lbl = Instance.new("TextLabel", outer)
    lbl.Size                   = UDim2.new(1,0,1,0)
    lbl.BackgroundTransparency = 1
    lbl.Font                   = Enum.Font.GothamBold
    lbl.TextScaled             = true
    lbl.TextStrokeTransparency = 0.35
    lbl.Text                   = ""
    lbl.TextColor3             = Color3.fromRGB(255,255,255)

    RunService.Heartbeat:Connect(function()
        if not S.KillerHUD then outer.Visible = false; return end
        local closest = math.huge
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LP and p.Character and Root then
                local r = p.Character:FindFirstChild("HumanoidRootPart")
                if r then
                    local d = (r.Position - Root.Position).Magnitude
                    if d < closest then closest = d end
                end
            end
        end
        if closest < 55 then
            outer.Visible = true
            local t = 1 - math.clamp(closest / 55, 0, 1)
            lbl.TextColor3       = Color3.fromRGB(255, math.floor(255*(1-t)), math.floor(255*(1-t)))
            outer.BackgroundColor3 = Color3.fromRGB(math.floor(160*t+10), 5, 5)
            lbl.Text = "⚠  KILLER NEARBY  —  " .. math.floor(closest) .. "m"
        else
            outer.Visible = false
        end
    end)
end

RunService.Heartbeat:Connect(function()
    if not Char or not Hum or not Root then return end

    if S.InfiniteStamina then
        for _, src in ipairs({LP, Char}) do
            for _, v in ipairs(src:GetChildren()) do
                if (v:IsA("NumberValue") or v:IsA("IntValue")) then
                    local n = v.Name:lower()
                    if n:find("stamina") or n:find("energy") or n:find("sprint") then
                        v.Value = S.MaxStamina
                    end
                end
            end
        end
        pcall(function()
            for _, st in ipairs({"leaderstats","Stats","PlayerStats"}) do
                local s = LP:FindFirstChild(st)
                if s then
                    for _, v in ipairs(s:GetChildren()) do
                        local n = v.Name:lower()
                        if n:find("stamina") or n:find("energy") then
                            v.Value = S.MaxStamina
                        end
                    end
                end
            end
        end)
    end

    pcall(function() Hum.WalkSpeed = S.WalkSpeed end)

    if S.UnlimitedZoom then
        LP.CameraMaxZoomDistance = 999
        LP.CameraMinZoomDistance = 0
    end

    if S.NoFog then
        Lighting.FogEnd   = 1e6
        Lighting.FogStart = 1e6
    end

    if S.AntiRagdoll then
        pcall(function()
            Hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll,    false)
            Hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, false)
            Hum:SetStateEnabled(Enum.HumanoidStateType.Physics,     false)
            local st = Hum:GetState()
            if st == Enum.HumanoidStateType.Ragdoll or st == Enum.HumanoidStateType.FallingDown then
                Hum:ChangeState(Enum.HumanoidStateType.GettingUp)
            end
        end)
    end

    if S.HitboxExpander then
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LP and p.Character then
                local r = p.Character:FindFirstChild("HumanoidRootPart")
                if r then
                    r.Size        = Vector3.new(S.HitboxSize, S.HitboxSize, S.HitboxSize)
                    r.Transparency = 0.85
                end
            end
        end
    end

    if S.TpAwayActive then
        for _, p in ipairs(Players:GetPlayers()) do
            if p ~= LP and p.Character then
                local r = p.Character:FindFirstChild("HumanoidRootPart")
                if r and (r.Position - Root.Position).Magnitude < S.AutoTpAway then
                    local a = math.random() * math.pi * 2
                    local d = S.AutoTpAway + math.random(10, 35)
                    SafeTP(Root.Position + Vector3.new(math.cos(a)*d, 0, math.sin(a)*d))
                end
            end
        end
    end
end)
