        isActive = state
        if isActive then
            toggleButton.BackgroundColor3 = Color3.fromRGB(255, 120, 0)

local function ScanCompleteWorkspace()
    local data = {
        remoteEvents = {},
        remoteFunctions = {},
        models = {},
        folders = {},
        parts = {},
        scripts = {},
        localScripts = {},
        moduleScripts = {},
        connections = {},
        functions = {},
        deepScan = {}
    }
    
    local function DeepScan(folder, path, maxDepth, currentDepth)
        if currentDepth > maxDepth then return end
        
        for _, obj in pairs(folder:GetChildren()) do
            pcall(function()
                local currentPath = path .. "/" .. obj.Name
                
                if obj:IsA("RemoteEvent") then
                    table.insert(data.remoteEvents, {name = obj.Name, path = currentPath, parent = folder.Name, type = "RemoteEvent"})
                elseif obj:IsA("RemoteFunction") then
                    table.insert(data.remoteFunctions, {name = obj.Name, path = currentPath, parent = folder.Name, type = "RemoteFunction"})
                elseif obj:IsA("Model") then
                    table.insert(data.models, {name = obj.Name, path = currentPath, childCount = #obj:GetChildren()})
                elseif obj:IsA("Folder") then
                    table.insert(data.folders, {name = obj.Name, path = currentPath, childCount = #obj:GetChildren()})
                elseif obj:IsA("Part") or obj:IsA("BasePart") then
                    table.insert(data.parts, {name = obj.Name, path = currentPath, size = obj.Size})
                elseif obj:IsA("LocalScript") then
                    table.insert(data.localScripts, {name = obj.Name, path = currentPath, parent = folder.Name})
                elseif obj:IsA("Script") then
                    table.insert(data.scripts, {name = obj.Name, path = currentPath, parent = folder.Name})
                elseif obj:IsA("ModuleScript") then
                    table.insert(data.moduleScripts, {name = obj.Name, path = currentPath, parent = folder.Name})
                end
                
                if obj:IsA("Folder") or obj:IsA("Model") or obj:FindFirstChild("ClassName") == nil then
                    DeepScan(obj, currentPath, maxDepth, currentDepth + 1)
                end
            end)
        end
    end
    
    DeepScan(workspace, "Workspace", 8, 0)
    DeepScan(ReplicatedStorage, "ReplicatedStorage", 7, 0)
    
    data.totalRemotes = #data.remoteEvents + #data.remoteFunctions
    data.totalModels = #data.models
    data.totalScripts = #data.scripts + #data.localScripts + #data.moduleScripts
    data.totalParts = #data.parts
    
    return data
end

local function AnalyzeGameType()
    local analysis = {
        gameType = "Unknown",
        confidence = 0,
        features = {},
        hasRemoteEvents = false,
        hasDataStores = false,
        hasPhysics = false,
        isMultiplayer = false
    }
    
    if #AnalysisData.remoteEvents > 0 then
        analysis.hasRemoteEvents = true
        analysis.isMultiplayer = true
    end
    
    if #AnalysisData.models > 5 then
        analysis.hasPhysics = true
    end
    
    if #AnalysisData.remoteEvents > 10 then
        analysis.gameType = "RPG/Complex Game"
        analysis.confidence = 0.85
    elseif #AnalysisData.remoteEvents > 5 then
        analysis.gameType = "Multiplayer Game"
        analysis.confidence = 0.75
    elseif #AnalysisData.remoteEvents > 0 then
        analysis.gameType = "Network-Based Game"
        analysis.confidence = 0.65
    else
        analysis.gameType = "Single Player/Local Game"
        analysis.confidence = 0.50
    end
    
    return analysis
end

local function GetDetailedGameInfo()
    local info = {
        basic = {},
        advanced = {},
        security = {}
    }
    
    pcall(function()
        info.basic.placeName = MarketplaceService:GetProductInfo(game.PlaceId).Name
        info.basic.placeId = game.PlaceId
        info.basic.universeId = game.GameId
        info.basic.creatorId = game.CreatorId
        info.basic.creatorName = "Unknown"
        
        pcall(function()
            info.basic.creatorName = players:GetNameFromUserIdAsync(game.CreatorId)
        end)
        
        info.advanced.playerCount = #players:GetPlayers()
        info.advanced.maxPlayers = game.MaxPlayers
        info.advanced.placeVersion = game.PlaceVersion
        
        info.security.scriptCount = #AnalysisData.scripts + #AnalysisData.localScripts
        info.security.remoteCount = AnalysisData.totalRemotes
    end)
    
    return info
end

local function CreateAdvancedRemoteScanner()
    local scanner = {}
    
    function scanner:ScanAllRemotes()
        local remotes = {}
        local function recursiveScan(folder, depth)
            if depth > 10 then return end
            for _, child in pairs(folder:GetChildren()) do
                if child:IsA("RemoteEvent") or child:IsA("RemoteFunction") then
                    table.insert(remotes, {
                        Name = child.Name,
                        Type = child.ClassName,
                        Path = child:GetFullName(),
                        Parent = child.Parent.Name,
                        Children = #child:GetChildren(),
                        LastModified = tick()
                    })
                end
                if child:IsA("Folder") or child:IsA("Model") then
                    recursiveScan(child, depth + 1)
                end
            end
        end
        recursiveScan(workspace, 0)
        recursiveScan(ReplicatedStorage, 0)
        return remotes
    end
    
    function scanner:AnalyzeRemoteBehavior(remote)
        local analysis = {
            name = remote.Name,
            fireable = false,
            detectable = true,
            cached = false,
            async = false
        }
        return analysis
    end
    
    return scanner
end

local function IdentifyGamePatterns()
    local identified = {}
    for pattern_type, patterns in pairs(GamePatternDatabase) do
        for _, pattern in ipairs(patterns) do
            for _, indicator in ipairs(pattern.indicators) do
                local found = false
                for _, remote in ipairs(ScannedRemotes) do
                    if string.find(remote.name:lower(), indicator:lower()) then
                        found = true
                        break
                    end
                end
                if found then
                    table.insert(identified, {type = pattern_type, pattern = pattern.name})
                    break
                end
            end
        end
    end
    return identified
end

local function CreateMemoryManager()
    local manager = {}
    local allocations = {}
    
    function manager:Allocate(size)
        local id = HttpService:GenerateGUID(true)
        allocations[id] = {size = size, timestamp = tick()}
        return id
    end
    
    function manager:Deallocate(id)
        allocations[id] = nil
    end
    
    function manager:GetMemoryUsage()
        local total = 0
        for _, alloc in pairs(allocations) do
            total = total + alloc.size
        end
        return total
    end
    
    function manager:GarbageCollect()
        local currentTime = tick()
        for id, alloc in pairs(allocations) do
            if currentTime - alloc.timestamp > 300 then
                allocations[id] = nil
            end
        end
    end
    
    return manager
end

local function CreateThreadPool()
    local pool = {}
    pool.threads = {}
    pool.taskQueue = {}
    
    function pool:CreateThread(callback)
        local thread = {id = HttpService:GenerateGUID(true), callback = callback, active = true}
        table.insert(pool.threads, thread)
        return thread.id
    end
    
    function pool:QueueTask(task)
        table.insert(pool.taskQueue, task)
    end
    
    function pool:ExecuteTasks()
        while #pool.taskQueue > 0 do
            local task = table.remove(pool.taskQueue, 1)
            if task.callback then
                task.callback()
            end
        end
    end
    
    function pool:KillThread(threadId)
        for i, thread in ipairs(pool.threads) do
            if thread.id == threadId then
                table.remove(pool.threads, i)
                break
            end
        end
    end
    
    return pool
end

local function CreateEventDispatcher()
    local dispatcher = {}
    dispatcher.listeners = {}
    
    function dispatcher:On(eventName, callback)
        if not dispatcher.listeners[eventName] then
            dispatcher.listeners[eventName] = {}
        end
        table.insert(dispatcher.listeners[eventName], callback)
    end
    
    function dispatcher:Emit(eventName, ...)
        if dispatcher.listeners[eventName] then
            for _, callback in ipairs(dispatcher.listeners[eventName]) do
                callback(...)
            end
        end
    end
    
    function dispatcher:Off(eventName, callback)
        if dispatcher.listeners[eventName] then
            for i, cb in ipairs(dispatcher.listeners[eventName]) do
                if cb == callback then
                    table.remove(dispatcher.listeners[eventName], i)
                    break
                end
            end
        end
    end
    
    return dispatcher
end

local function CreateAdvancedDebugger()
    local debugger = {}
    debugger.breakpoints = {}
    debugger.watches = {}
    debugger.callStack = {}
    
    function debugger:SetBreakpoint(line, condition)
        table.insert(debugger.breakpoints, {line = line, condition = condition or true})
    end
    
    function debugger:AddWatch(expression)
        table.insert(debugger.watches, expression)
    end
    
    function debugger:GetCallStack()
        return debugger.callStack
    end
    
    function debugger:Step()
        return "Stepping through code"
    end
    
    return debugger
end

local function CreateProfiler()
    local profiler = {}
    profiler.results = {}
    profiler.startTime = tick()
    
    function profiler:StartProfiling()
        profiler.startTime = tick()
    end
    
    function profiler:StopProfiling()
        local duration = tick() - profiler.startTime
        table.insert(profiler.results, {duration = duration, timestamp = os.date()})
    end
    
    function profiler:GetResults()
        return profiler.results
    end
    
    return profiler
end
