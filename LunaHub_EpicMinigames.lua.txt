local function Notify(title, msg, dur)
    Safe(StarterGui.SetCore, StarterGui, "SendNotification", {Title=title, Text=msg, Duration=dur or 4})

local function FindRemote(name)
    for _, v in ipairs(ReplicatedStorage:GetDescendants()) do
        if v.Name:lower():find(name:lower()) then return v end
    end
    for _, v in ipairs(workspace:GetDescendants()) do
        if v.Name:lower():find(name:lower()) then return v end
    end
    return nil
end

local function Fire(name, ...)
    local r = FindRemote(name)
    if not r then return end
    if r:IsA("RemoteEvent") then Safe(function() r:FireServer(...) end) end
    if r:IsA("RemoteFunction") then Safe(function() r:InvokeServer(...) end) end
end

local function DoCustomScore(amt)
    local names = {"SetScore","AddScore","GiveCoins","AddCoins","Points","AddPoints","SetPoints","SetMoney","AddMoney"}
    for _, n in ipairs(names) do Fire(n, amt) end
    local stats = LP:FindFirstChild("leaderstats") or LP:FindFirstChild("Stats")
    if stats then
        for _, v in ipairs(stats:GetChildren()) do
            if v:IsA("IntValue") or v:IsA("NumberValue") then
                Safe(function() v.Value = amt end)
            end
        end
    end
end

local function DoAlwaysPlaying()
    local evs = {"PlayerJoined","JoinGame","JoinRound","RejoinRound","ForceJoin","AddPlayer","StartPlaying"}
    for _, n in ipairs(evs) do Fire(n, LP) end
    for _, tag in ipairs({"Playing","InGame","InRound","IsPlaying"}) do
        if LP and not LP:FindFirstChild(tag) then
            local v = Instance.new("BoolValue")
            v.Name = tag; v.Value = true; v.Parent = LP
        end
    end
end

local function DoCheatRound()
    if not Char or not Hum then return end
    Hum.MaxHealth = math.huge
    Hum.Health = math.huge
    Hum:SetAttribute("Invincible", true)
    for _, tag in ipairs({"GodMode","Invincible","Protected","Immune","Safe","NoKill"}) do
        if not Char:FindFirstChild(tag) then
            local v = Instance.new("BoolValue")
            v.Name = tag; v.Value = true; v.Parent = Char
        end
    end
    Fire("GodMode")
    Fire("Protect", LP)
    Fire("SetInvincible", LP, true)
end

local function DoAutoWin()
    for _, n in ipairs({"Win","RoundWin","PlayerWon","AwardWin","FinishRound","CompleteRound","EndRound","SetWinner"}) do
        Fire(n, LP)
    end
    for _, tag in ipairs({"Winner","Win","Finish","Complete"}) do

            local v = Instance.new("BoolValue")

end

local function DoInfStamina()
    for _, v in ipairs(LP:GetDescendants()) do
        if (v.Name:lower():find("stam") or v.Name:lower():find("energy")) and
           (v:IsA("NumberValue") or v:IsA("IntValue")) then
            if v.Value < 50 then v.Value = 100 end
        end
    end
end

local function EnableNoClip()
    NoClipConn = RunService.Stepped:Connect(function()
        if not ST.NoClip or not Char then return end
        for _, p in ipairs(Char:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide = false end
        end
    end)
end

local function DisableNoClip()
    if NoClipConn then NoClipConn:Disconnect(); NoClipConn = nil end
    if Char then
        for _, p in ipairs(Char:GetDescendants()) do
            if p:IsA("BasePart") then p.CanCollide = true end
        end
    end
end

local function StartFly()
    if not HRP then return end
    local bv = Instance.new("BodyVelocity")
    bv.MaxForce = Vector3.new(1e9, 1e9, 1e9)
    bv.Velocity = Vector3.zero
    bv.Parent = HRP
    local bg = Instance.new("BodyGyro")
    bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
    bg.CFrame = HRP.CFrame
    bg.Parent = HRP
    FlyBodies = {bv, bg}
    FlyConn = RunService.Heartbeat:Connect(function()
        if not ST.Fly then return end
        local cf = Cam.CFrame
        local dir = Vector3.zero
        if FlyCtrl.F == 1 then dir += cf.LookVector end
        if FlyCtrl.B == 1 then dir -= cf.LookVector end
        if FlyCtrl.L == 1 then dir -= cf.RightVector end
        if FlyCtrl.R == 1 then dir += cf.RightVector end
        bv.Velocity = dir.Magnitude > 0 and dir.Unit * ST.FlySpeed or Vector3.zero
        bg.CFrame = cf
    end)
end

local function StopFly()
    if FlyConn then FlyConn:Disconnect(); FlyConn = nil end
    for _, b in ipairs(FlyBodies) do Safe(function() b:Destroy() end) end
    FlyBodies = {}
end

local function CreateESP(player)
    if player == LP then return end
    Safe(function()
        local bill = Instance.new("BillboardGui")
        bill.AlwaysOnTop = true
        bill.Size = UDim2.new(0, 130, 0, 32)
        bill.StudsOffset = Vector3.new(0, 4, 0)
        bill.Adornee = nil

        local box = Instance.new("Frame", bill)
        box.Size = UDim2.new(1, 0, 1, 0)
        box.BackgroundColor3 = Color3.fromRGB(20, 10, 35)
        box.BackgroundTransparency = 0.3
        box.BorderSizePixel = 0

        local lbl = Instance.new("TextLabel", box)
        lbl.Size = UDim2.new(1, 0, 1, 0)
        lbl.BackgroundTransparency = 1
        lbl.Text = player.Name
        lbl.TextColor3 = Color3.fromRGB(200, 130, 255)
        lbl.TextStrokeTransparency = 0
        lbl.Font = Enum.Font.GothamBold
        lbl.TextSize = 13

        local function Attach(char)
            local hrp = char:WaitForChild("HumanoidRootPart", 5)
            if hrp then bill.Parent = hrp end
        end

        if player.Character then Attach(player.Character) end
        player.CharacterAdded:Connect(Attach)
        ESPCache[player.Name] = bill
    end)
end

local function RemoveESP(player)
    if ESPCache[player.Name] then
        Safe(function() ESPCache[player.Name]:Destroy() end)
        ESPCache[player.Name] = nil
    end
end

local function EnableESP()
    for _, p in ipairs(Players:GetPlayers()) do CreateESP(p) end
    Players.PlayerAdded:Connect(CreateESP)
    Players.PlayerRemoving:Connect(RemoveESP)
end

local function DisableESP()
    for _, g in pairs(ESPCache) do Safe(function() g:Destroy() end) end
    ESPCache = {}
end

local ADMIN_NAMES = {
    "tcgnry","TwentyTwoPilots","Zomebody","Stickmasterluke","Roblox",
    "builderman","Admin","Moderator","Staff","Owner","Developer",
}
local function IsAdmin(player)
    local name = player.Name:lower()
    for _, a in ipairs(ADMIN_NAMES) do
        if name:find(a:lower()) then return true end
    end
    return player:GetRankInGroup(1200769) > 0
end

local function StartAdminAlert()
    for _, p in ipairs(Players:GetPlayers()) do
        if p ~= LP then
            Safe(function()
                if IsAdmin(p) then
                    Notify("⚠️ Luna Hub", "Admin detected: " .. p.Name, 8)
                end
            end)
        end
    end
    AdminConn = Players.PlayerAdded:Connect(function(p)
        if not ST.AdminAlert then return end
        task.wait(1)
        Safe(function()
            if IsAdmin(p) then
                Notify("⚠️ Luna Hub", "Admin joined: " .. p.Name, 8)
            end
        end)
    end)
end

local function DoServerHop()
    local placeId = game.PlaceId
    local ok, data = Safe(function()
        return HttpService:JSONDecode(
            game:HttpGet("https://games.roblox.com/v1/games/"..placeId.."/servers/Public?sortOrder=Asc&limit=100")
        )
    end)
    if not ok or not data then return end
    for _, srv in ipairs(data.data or {}) do
        if srv.id ~= game.JobId and srv.playing < srv.maxPlayers then
            TeleportService:TeleportToPlaceInstance(placeId, srv.id, LP)
            return
        end
    end
    Notify("Luna Hub", "No available servers found.", 4)
end

local function MakeSlider(parent, txt, key, min, max, order, cb)
    local f = Instance.new("Frame", parent)
    f.Size = UDim2.new(1, -4, 0, 58)
    f.BackgroundColor3 = BG_CARD
    f.BorderSizePixel = 0
    f.LayoutOrder = order or 0
    Instance.new("UICorner", f).CornerRadius = UDim.new(0, 8)

    local lbl = Instance.new("TextLabel", f)
    lbl.Size = UDim2.new(1, -10, 0, 22)
    lbl.Position = UDim2.new(0, 10, 0, 4)
    lbl.BackgroundTransparency = 1
    lbl.Text = txt .. ": " .. ST[key]
    lbl.TextColor3 = TXT_WHITE
    lbl.Font = Enum.Font.Gotham
    lbl.TextSize = 12
    lbl.TextXAlignment = Enum.TextXAlignment.Left

    local track = Instance.new("Frame", f)
    track.Size = UDim2.new(1, -20, 0, 6)
    track.Position = UDim2.new(0, 10, 0, 38)
    track.BackgroundColor3 = BG_PANEL
    track.BorderSizePixel = 0
    Instance.new("UICorner", track).CornerRadius = UDim.new(0, 3)

    local pct = math.clamp((ST[key] - min) / (max - min), 0, 1)
    local fill = Instance.new("Frame", track)
    fill.Size = UDim2.new(pct, 0, 1, 0)
    fill.BackgroundColor3 = AC_PURPLE
    fill.BorderSizePixel = 0
    Instance.new("UICorner", fill).CornerRadius = UDim.new(0, 3)

    local knob = Instance.new("Frame", track)
    knob.Size = UDim2.new(0, 12, 0, 12)
    knob.Position = UDim2.new(pct, -6, 0.5, -6)
    knob.BackgroundColor3 = AC_PINK
    knob.BorderSizePixel = 0
    Instance.new("UICorner", knob).CornerRadius = UDim.new(1, 0)

    local dragging = false
    track.InputBegan:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.MouseButton1 or
           inp.UserInputType == Enum.UserInputType.Touch then
            dragging = true
        end
    end)
    UserInputService.InputEnded:Connect(function(inp)
        if inp.UserInputType == Enum.UserInputType.MouseButton1 or
           inp.UserInputType == Enum.UserInputType.Touch then
            dragging = false
        end
    end)
    UserInputService.InputChanged:Connect(function(inp)
        if not dragging then return end
        if inp.UserInputType ~= Enum.UserInputType.MouseMovement and
           inp.UserInputType ~= Enum.UserInputType.Touch then return end
        local rel = math.clamp((inp.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
        local val = math.floor(min + (max - min) * rel)
        ST[key] = val
        fill.Size = UDim2.new(rel, 0, 1, 0)
        knob.Position = UDim2.new(rel, -6, 0.5, -6)
        lbl.Text = txt .. ": " .. val
        if cb then cb(val) end
    end)
    return f
end

UserInputService.InputBegan:Connect(function(inp, g)
    if g then return end
    if inp.KeyCode == Enum.KeyCode.Insert then GUI.Enabled = not GUI.Enabled end
    if inp.KeyCode == Enum.KeyCode.F then
        ST.Fly = not ST.Fly
        if ST.Fly then StartFly() else StopFly() end
    end
    if inp.KeyCode == Enum.KeyCode.E and ST.ClickTP and HRP then
        local pos = Mouse.Hit.Position
        if pos then HRP.CFrame = CFrame.new(pos + Vector3.new(0, 4, 0)) end
    end
    if inp.KeyCode == Enum.KeyCode.Delete then task.spawn(DoServerHop) end
    if inp.KeyCode == Enum.KeyCode.W then FlyCtrl.F = 1 end
    if inp.KeyCode == Enum.KeyCode.S then FlyCtrl.B = 1 end
    if inp.KeyCode == Enum.KeyCode.A then FlyCtrl.L = 1 end
    if inp.KeyCode == Enum.KeyCode.D then FlyCtrl.R = 1 end
end)

end)
