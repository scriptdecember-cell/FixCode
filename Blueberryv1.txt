local function CreateToggle(parent, name, defaultState, callback)
    local container = Instance.new("Frame")
    container.Size = UDim2.new(1, -12, 0, 38)
    container.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    container.BorderColor3 = Color3.fromRGB(70, 70, 70)
    container.BorderSizePixel = 1
    container.Parent = parent
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 6)
    corner.Parent = container
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(1, -55, 1, 0)
    label.BackgroundTransparency = 1
    label.TextColor3 = Color3.fromRGB(200, 200, 200)
    label.TextSize = 13
    label.Text = name
    label.Font = Enum.Font.Gotham
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.Parent = container
    
    local labelPad = Instance.new("UIPadding")
    labelPad.PaddingLeft = UDim.new(0, 12)
    labelPad.Parent = label
    
    local toggleButton = Instance.new("TextButton")
    toggleButton.Size = UDim2.new(0, 48, 0, 26)
    toggleButton.Position = UDim2.new(1, -54, 0.5, -13)
    toggleButton.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
    toggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    toggleButton.Text = "OFF"
    toggleButton.TextSize = 11
    toggleButton.Font = Enum.Font.GothamBold
    toggleButton.BorderSizePixel = 0
    toggleButton.Parent = container
    
    local btnCorner = Instance.new("UICorner")
    btnCorner.CornerRadius = UDim.new(0, 5)
    btnCorner.Parent = toggleButton
    
    local isActive = defaultState or false
    
    local function UpdateToggle(state)
        isActive = state
        if isActive then
            toggleButton.BackgroundColor3 = Color3.fromRGB(0, 150, 255)
            toggleButton.Text = "ON"
        else
            toggleButton.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
            toggleButton.Text = "OFF"
        end
        callback(isActive)
    end
    
    toggleButton.MouseButton1Click:Connect(function()
        UpdateToggle(not isActive)
    end)
    
    toggleButton.MouseEnter:Connect(function()
        if isActive then
            toggleButton.BackgroundColor3 = Color3.fromRGB(0, 170, 255)
        else
            toggleButton.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
        end
    end)
    
    toggleButton.MouseLeave:Connect(function()
        if isActive then
            toggleButton.BackgroundColor3 = Color3.fromRGB(0, 150, 255)
        else
            toggleButton.BackgroundColor3 = Color3.fromRGB(80, 80, 80)
        end
    end)
    
    return {button = toggleButton, getValue = function() return isActive end}
end

local function CreateSlider(parent, name, min, max, default, callback)
    local container = Instance.new("Frame")
    container.Size = UDim2.new(1, -12, 0, 55)
    container.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
    container.BorderColor3 = Color3.fromRGB(70, 70, 70)
    container.BorderSizePixel = 1
    container.Parent = parent

    corner.CornerRadius = UDim.new(0, 6)
    corner.Parent = container
    
    local label = Instance.new("TextLabel")
    label.Size = UDim2.new(0.6, 0, 0, 20)
    label.Position = UDim2.new(0, 10, 0, 5)
    label.BackgroundTransparency = 1
    label.TextColor3 = Color3.fromRGB(200, 200, 200)
    label.TextSize = 12
    label.Text = name
    label.Font = Enum.Font.Gotham
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.Parent = container
    
    local valueLabel = Instance.new("TextLabel")
    valueLabel.Size = UDim2.new(0.4, -10, 0, 20)
    valueLabel.Position = UDim2.new(0.6, 0, 0, 5)
    valueLabel.BackgroundTransparency = 1
    valueLabel.TextColor3 = Color3.fromRGB(0, 150, 255)
    valueLabel.TextSize = 13
    valueLabel.Text = tostring(math.floor(default))
    valueLabel.Font = Enum.Font.GothamBold
    valueLabel.TextXAlignment = Enum.TextXAlignment.Right
    valueLabel.Parent = container
    
    local sliderBg = Instance.new("Frame")
    sliderBg.Size = UDim2.new(1, -20, 0, 8)
    sliderBg.Position = UDim2.new(0, 10, 0, 32)
    sliderBg.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    sliderBg.BorderColor3 = Color3.fromRGB(80, 80, 80)
    sliderBg.BorderSizePixel = 1
    sliderBg.Parent = container
    
    local sliderFill = Instance.new("Frame")
    sliderFill.Size = UDim2.new((default - min) / (max - min), 0, 1, 0)
    sliderFill.BackgroundColor3 = Color3.fromRGB(0, 150, 255)
    sliderFill.BorderSizePixel = 0
    sliderFill.Parent = sliderBg
    
    local function updateValue(value)
        value = math.max(min, math.min(max, value))
        valueLabel.Text = tostring(math.floor(value))
        sliderFill.Size = UDim2.new((value - min) / (max - min), 0, 1, 0)
        callback(value)
    end
    
    local dragging = false
    sliderBg.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
        end
    end)
    
    sliderBg.InputEnded:Connect(function(input, gameProcessed)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input, gameProcessed)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local mouse = localPlayer:GetMouse()
            local relativePos = mouse.X - sliderBg.AbsolutePosition.X
            local percentage = math.max(0, math.min(1, relativePos / sliderBg.AbsoluteSize.X))
            updateValue(min + (max - min) * percentage)
        end
    end)
end

local function startAutoParry()
    ballsFolder = workspace:FindFirstChild("Balls")
    
    if not ballsFolder then
        print("Balls folder not found!")
        return
    end
    
    parryRemote = replicatedStorage:FindFirstChild("Remotes") and replicatedStorage.Remotes:FindFirstChild("ParryButtonPress")
    abilityRemote = replicatedStorage:FindFirstChild("Remotes") and replicatedStorage.Remotes:FindFirstChild("AbilityButtonPress")

    if not parryRemote then
        print("ParryButtonPress remote not found!")
        return
    end

    print("Auto Parry started!")

    local function chooseNewFocusedBall()
        local balls = ballsFolder:GetChildren()
        local closestBall = nil
        local closestDistance = math.huge
        
        for _, ball in ipairs(balls) do
            pcall(function()
                if ball:IsA("BasePart") then
                    local isRealBall = ball:GetAttribute("realBall")
                    
                    if isRealBall == true then
                        local distance = (ball.Position - character.HumanoidRootPart.Position).Magnitude
                        if distance < closestDistance then
                            closestBall = ball
                            closestDistance = distance
                        end
                    end
                end
            end)
        end
        
        focusedBall = closestBall
    end

    local function getDynamicThreshold(ballVelocityMagnitude)
        if ballVelocityMagnitude > 60 then
            return math.max(0.20, Config.BaseThreshold - (ballVelocityMagnitude * Config.VelocityScalingFast))
        else
            return math.min(0.01, Config.BaseThreshold + (ballVelocityMagnitude * Config.VelocityScalingSlow))
        end
    end

    local function timeUntilImpact(ballVelocity, distanceToPlayer, playerVelocity)
        if not focusedBall then return math.huge end
        
        local directionToPlayer = (character.HumanoidRootPart.Position - focusedBall.Position).Unit
        local velocityTowardsPlayer = ballVelocity:Dot(directionToPlayer) - playerVelocity:Dot(directionToPlayer)
        
        if velocityTowardsPlayer <= 0 then
            return math.huge
        end
        
        return (distanceToPlayer - Config.ParryDistance) / velocityTowardsPlayer
    end

    local function checkBallDistance()
        pcall(function()
            if not character or not character:FindFirstChild("HumanoidRootPart") then return end

            local charPos = character.HumanoidRootPart.Position
            local charVel = character.HumanoidRootPart.Velocity

            if focusedBall and (not focusedBall.Parent or not focusedBall:IsDescendantOf(workspace)) then
                chooseNewFocusedBall()
            end
            
            if not focusedBall then 
                chooseNewFocusedBall()
            end

            local ball = focusedBall
            if not ball or not ball.Parent then return end
            
            local distanceToPlayer = (ball.Position - charPos).Magnitude
            local ballVelocityTowardsPlayer = ball.Velocity:Dot((charPos - ball.Position).Unit)
            
            if distanceToPlayer < 10 then
                parryRemote:Fire()
                return
            end

            if timeUntilImpact(ball.Velocity, distanceToPlayer, charVel) < getDynamicThreshold(ballVelocityTowardsPlayer) then
                parryRemote:Fire()
            end
        end)
    end

    if heartbeatConnection then
        heartbeatConnection:Disconnect()
    end
    
    heartbeatConnection = runService.Heartbeat:Connect(function()
        checkBallDistance()
    end)
end

local function stopAutoParry()
    if heartbeatConnection then
        heartbeatConnection:Disconnect()
        heartbeatConnection = nil
    end
end

local function startStaySpawn()
    if staySpawnConnection then
        staySpawnConnection:Disconnect()
    end
    
    staySpawnConnection = runService.Heartbeat:Connect(function()
        if Config.StaySpawnEnabled and character then
            pcall(function()
                character.Parent = workspace.Alive
            end)
        end
    end)
end

local function stopStaySpawn()
    if staySpawnConnection then
        staySpawnConnection:Disconnect()
        staySpawnConnection = nil
    end
end
